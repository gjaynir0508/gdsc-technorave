def mystery_function(input_string):
    return ''.join([char.upper() if char.islower() else char.lower() for char in input_string])

# Use the mystery function to decode the message
encoded_message = "rNxNYYz oB Nbyg"
decoded_message = mystery_function(encoded_message)
